本番環境と異なり、PDRのST環境は以下の設定
・customerSeqのMAXVALUE 799999⇒999999
・orderNoSeqのMINVALUE 6000000⇒1000000


/* CUS01 顧客番号SEQ */
CREATE SEQUENCE customerSeq INCREMENT 1 MINVALUE 700000 MAXVALUE 999999 START 700001;
/* CUS02 受注番号SEQ */
CREATE SEQUENCE orderNoSeq INCREMENT 1 MINVALUE 1000000 MAXVALUE 9999999 START 6000001;
